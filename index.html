// ── State ──────────────────────────────────────────────────────────────

let questions = [], currentIdx = 0, score = 0, missed = [], answered = false;

let weakStats = {}; // { category: { correct, total } }


const CATEGORIES = ['time', 'place', 'collocations', 'phrasal-verbs', 'cause-result'];


// ── LocalStorage helpers ───────────────────────────────────────────────

function load(key, fallback) {

  try { const v = localStorage.getItem(key); return v !== null ? JSON.parse(v) : fallback; }

  catch { return fallback; }

}

function save(key, val) { try { localStorage.setItem(key, JSON.stringify(val)); } catch {} }


function updateStreakUI() {

  const streak = load('streak', 0);

  const best = load('best', null);

  const ws = load('weakStats', {});

  weakStats = ws;


  document.getElementById('streakCount').textContent = streak;

  document.getElementById('bestScore').textContent = best !== null ? best + '/20' : '—';


  // Determine weakest category

  let worstCat = 'balanced', worstRate = 1;

  for (const [cat, d] of Object.entries(ws)) {

    if (d.total >= 3) {

      const rate = d.correct / d.total;

      if (rate < worstRate) { worstRate = rate; worstCat = cat; }

    }

  }

  document.getElementById('focusArea').textContent = worstCat;

  return { worstCat, worstRate, ws };

}


function bumpStreak() {

  const last = load('lastDate', '');

  const today = new Date().toDateString();

  const yesterday = new Date(Date.now() - 86400000).toDateString();

  let streak = load('streak', 0);

  if (last === today) {}

  else if (last === yesterday) { streak++; }

  else { streak = 1; }

  save('streak', streak);

  save('lastDate', today);

}


function saveSessionScore(s) {

  const best = load('best', 0);

  if (s > best) save('best', s);

}


function saveWeakStats() {

  save('weakStats', weakStats);

}


// ── Build Claude prompt ────────────────────────────────────────────────

function buildPrompt() {

  const ws = load('weakStats', {});

  const sessionCount = load('sessionCount', 0);


  // Work out category weights

  let weights = {};

  CATEGORIES.forEach(c => {

    const d = ws[c] || { correct: 0, total: 0 };

    // More total attempts + lower accuracy = higher weight

    const accuracy = d.total > 0 ? d.correct / d.total : 0.6;

    weights[c] = Math.round((1 - accuracy) * 10 + 2);

  });


  const weightStr = CATEGORIES.map(c => `${c}: weight ${weights[c]}/10`).join(', ');

  const totalSessions = sessionCount + 1;


  return `You are generating a daily English preposition practice quiz for an intermediate learner (occasional mistakes, not a beginner).


Session number: ${totalSessions}

Category weakness weights (higher = needs more practice): ${weightStr}


Generate exactly 20 fill-in-the-blank questions. Distribute them roughly according to the weights above — categories with higher weights should appear more.


Each question must be a realistic, natural English sentence (professional/everyday context) with one blank where a preposition or prepositional phrase goes.


Return ONLY a valid JSON array of 20 objects. No markdown, no explanation, no preamble. Each object:

{

  "sentence": "She has been working here ___ 2019.",

  "answer": "since",

  "options": ["since", "for", "from", "in"],

  "category": "time",

  "rule": "Use 'since' for a specific point in time. 'For' marks a duration."

}


Rules:

- options must always include the correct answer + 3 plausible distractors

- sentences should feel natural, not textbook-stilted

- mix difficulty — some straightforward, some subtle

- categories must be one of: time, place, collocations, phrasal-verbs, cause-result

- rule must be one clear, concise sentence of explanation

- vary contexts: work emails, travel, daily life, relationships, news


Return only the JSON array.`;

}


// ── API call ───────────────────────────────────────────────────────────

async function generateQuestions(apiKey) {

  const response = await fetch('https://api.anthropic.com/v1/messages', {

    method: 'POST',

    headers: { 'Content-Type': 'application/json', 'x-api-key': apiKey, 'anthropic-version': '2023-06-01', 'anthropic-dangerous-direct-browser-access': 'true' },

    body: JSON.stringify({

      model: 'claude-sonnet-4-20250514',

      max_tokens: 4000,

      messages: [{ role: 'user', content: buildPrompt() }]

    })

  });


  if (!response.ok) {

    const err = await response.json().catch(() => ({}));

    throw new Error(err?.error?.message || `API error ${response.status}`);

  }


  const data = await response.json();

  const text = data.content.find(b => b.type === 'text')?.text || '';

  const clean = text.replace(/```json|```/g, '').trim();

  const parsed = JSON.parse(clean);

  if (!Array.isArray(parsed) || parsed.length < 10) throw new Error('Invalid questions format');

  return parsed.slice(0, 20);

}


// ── Session flow ───────────────────────────────────────────────────────

async function initSession() {

  const keyInput = document.getElementById('apiKeyInput');

  let apiKey = keyInput.value.trim() || load('apiKey', '');


  if (!apiKey) {

    showSetupError('Please enter your Anthropic API key.');

    return;

  }

  save('apiKey', apiKey);


  // Pre-fill for next time

  keyInput.value = apiKey;


  hide('setupCard'); hide('quizCard'); hide('progressWrap'); hide('resultsCard');

  show('loadingCard');


  const { worstCat } = updateStreakUI();

  document.getElementById('focusLabel').textContent =

    worstCat !== 'balanced' ? `🎯 Extra focus on: ${worstCat}` : 'Balanced practice — no weak spots yet';


  try {

    questions = await generateQuestions(apiKey);

  } catch (e) {

    hide('loadingCard'); show('setupCard');

    showSetupError('Could not generate questions: ' + e.message);

    return;

  }


  currentIdx = 0; score = 0; missed = []; answered = false;

  bumpStreak(); updateStreakUI();

  save('sessionCount', (load('sessionCount', 0) + 1));


  hide('loadingCard');

  show('progressWrap'); show('quizCard');

  renderQuestion();

}


function renderQuestion() {

  const q = questions[currentIdx];

  answered = false;


  document.getElementById('questionCounter').textContent = `Question ${currentIdx + 1} of 20`;

  document.getElementById('scoreTracker').textContent = `Score: ${score}`;

  document.getElementById('progressFill').style.width = `${((currentIdx + 1) / 20) * 100}%`;

  document.getElementById('categoryTag').textContent = q.category;

  document.getElementById('questionText').innerHTML = q.sentence.replace('___', '<span class="blank">___</span>');


  const fb = document.getElementById('feedback');

  fb.className = 'feedback'; fb.innerHTML = '';

  document.getElementById('nextBtn').style.display = 'none';


  const grid = document.getElementById('optionsGrid');

  grid.innerHTML = '';

  shuffle(q.options).forEach(opt => {

    const btn = document.createElement('button');

    btn.className = 'option-btn';

    btn.textContent = opt;

    btn.onclick = () => selectAnswer(opt, btn, q);

    grid.appendChild(btn);

  });

}


function selectAnswer(choice, btn, q) {

  if (answered) return;

  answered = true;


  // Update weak stats

  if (!weakStats[q.category]) weakStats[q.category] = { correct: 0, total: 0 };

  weakStats[q.category].total++;


  document.querySelectorAll('.option-btn').forEach(b => {

    b.disabled = true;

    if (b.textContent === q.answer) b.classList.add('reveal');

  });


  const fb = document.getElementById('feedback');

  const isCorrect = choice === q.answer;


  if (isCorrect) {

    btn.classList.add('correct');

    score++;

    weakStats[q.category].correct++;

    fb.className = 'feedback correct show';

    fb.innerHTML = `✓ Correct! <span class="rule">${q.rule}</span>`;

  } else {

    btn.classList.add('wrong');

    fb.className = 'feedback wrong show';

    fb.innerHTML = `✗ Answer: <strong>"${q.answer}"</strong>. <span class="rule">${q.rule}</span>`;

    missed.push({ q, yourAnswer: choice });

  }


  saveWeakStats();

  const nb = document.getElementById('nextBtn');

  nb.style.display = 'block';

  nb.textContent = currentIdx < questions.length - 1 ? 'Next →' : 'See Results';

}


function nextQuestion() {

  currentIdx++;

  if (currentIdx >= questions.length) showResults();

  else renderQuestion();

}


function showResults() {

  hide('progressWrap'); hide('quizCard');

  show('resultsCard');


  document.getElementById('finalScore').textContent = `${score} / 20`;

  const pct = (score / 20) * 100;

  document.getElementById('finalLabel').textContent =

    pct === 100 ? 'Perfect! 🌟' :

    pct >= 85 ? 'Excellent — you\'re getting sharp! 💪' :

    pct >= 70 ? 'Good work. Keep the streak going.' :

    pct >= 50 ? 'Decent start. Claude will target your weak spots next time.' :

    'Keep going — tomorrow\'s questions will focus on what you missed. 🔄';


  saveSessionScore(score);

  updateStreakUI();


  // Weak spots breakdown

  const ws = load('weakStats', {});

  let wsHtml = '<h3>All-time accuracy by category</h3>';

  CATEGORIES.forEach(cat => {

    const d = ws[cat] || { correct: 0, total: 0 };

    const pct = d.total > 0 ? Math.round((d.correct / d.total) * 100) : null;

    const barW = pct !== null ? (100 - pct) : 0;

    wsHtml += `<div class="weak-row">

      <span class="cat">${cat}</span>

      <div class="bar-wrap"><div class="bar-inner" style="width:${barW}%"></div></div>

      <span class="pct">${pct !== null ? pct + '%' : '—'}</span>

    </div>`;

  });

  document.getElementById('weakSpots').innerHTML = wsHtml;


  // Missed questions

  const ms = document.getElementById('missedSection');

  if (missed.length === 0) {

    ms.innerHTML = '<p style="color:#4ade80;text-align:center;margin-top:16px;">No mistakes today! 🏆</p>';

  } else {

    ms.innerHTML = `<h3>Review (${missed.length} missed)</h3>` + missed.map(m =>

      `<div class="missed-item">

        <div class="sentence">${m.q.sentence.replace('___', `<strong>${m.q.answer}</strong>`)}</div>

        <div class="answer">✓ ${m.q.answer}</div>

        <div class="yours">✗ You answered: ${m.yourAnswer}</div>

      </div>`

    ).join('');

  }

}


// ── Utilities ──────────────────────────────────────────────────────────

function shuffle(arr) {

  const a = [...arr];

  for (let i = a.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [a[i], a[j]] = [a[j], a[i]];

  }

  return a;

}


function show(id) { document.getElementById(id).style.display = 'block'; }

function hide(id) { document.getElementById(id).style.display = 'none'; }

function showSetupError(msg) {

  const el = document.getElementById('setupError');

  el.textContent = msg; el.style.display = 'block';

}


// ── Init ───────────────────────────────────────────────────────────────

const savedKey = load('apiKey', '');

if (savedKey) document.getElementById('apiKeyInput').value = savedKey;

updateStreakUI();
